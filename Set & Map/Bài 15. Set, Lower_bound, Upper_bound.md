package hoclaptrinhjava;
import java.util.*;

    public class Buoi1 {
        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
            int n = sc.nextInt();
            TreeMap<Integer, Integer> mp = new TreeMap<>();
            for(int i = 0; i < n; i++){
                int x = sc.nextInt();
                if(!mp.containsKey(x)){
                    mp.put(x,1);
                }else{
                    mp.put(x, mp.get(x) + 1);
                }
            }
            int q = sc.nextInt();
            while(q-- > 0){
                int tt = sc.nextInt();
                if(tt == 1){
                    int x = sc.nextInt();
                    if(!mp.containsKey(x)){
                        mp.put(x,1);
                    }else{
                        mp.put(x, mp.get(x) + 1);
                    }
                }else if(tt == 2){
                    int x = sc.nextInt();
                    if(mp.containsKey(x)){
                        int fre = mp.get(x);
                        if(fre > 1) mp.put(x, fre - 1);
                        else mp.remove(x);
                    }
                }else{
                    int x =sc.nextInt();
                    Integer res = mp.floorKey(x);
                    if(res != null) System.out.println(res);
                    else System.out.println(-1);
                }

            }
    }
}
